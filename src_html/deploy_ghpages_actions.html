Deploy project on GitHub Pages with GitHub Actions

Create GitHub Actions YAML file.

Will use James Ives‚Äô github-pages-deploy-action

# .github/workflows/gh_pages_deploy.yml
name: Build and Deploy
on:
  push:
    branches:
      - main
jobs:
  build-and-deploy:
    runs-on: ubuntu-latest
    steps:
      - name: Checkout üõéÔ∏è
        uses: actions/checkout@v2.3.1

      - name: Install and Build üîß # This example project is built using npm and outputs the result to the 'build' folder. Replace with the commands required to build your project, or remove this step entirely if your site is pre-built.
        run: |
          npm install
          npm run lint
          npm run build

      - name: Deploy üöÄ
        uses: JamesIves/github-pages-deploy-action@4.1.4
        with:
          branch: gh-pages # The branch the action should deploy to.
          folder: dist # The folder the action should deploy.

- name: Create .env File
  run: |
    touch .env
    echo API_KEY=${{ secrets.API_KEY }} >> .env
